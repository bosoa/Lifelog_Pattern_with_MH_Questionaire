```python
# 데이터 없는 센서 자동 필터링
def load_sensor_data(self, exclude_empty=True):
    sensor_data = {}
    for name, filename in self.sensor_files.items():
        df = pd.read_csv(self.data_dir / filename)
        
        # 데이터가 완전히 비어있는지 확인
        if exclude_empty:
            data_cols = df.columns[1:]
            if (df[data_cols] == 0).all().all():
                print(f"⚠️  '{name}' 센서는 데이터가 없어 제외됩니다.")
                continue
        
        sensor_data[name] = df
    return sensor_data
```
